#製造平均數兩種寫法
#很直覺，不過如果有遺漏值就麻煩了
dta$功能 <-(dta$視力 + dta$聽力 + dta$移動 + dta$溝通)/4

#將五個變數以兩個類別變項分組,取平均
hlt <- aggregate(cbind(視力,聽力,移動,溝通,功能) ~性別+年齡,data = dta, mean)

#把 hlth 的前兩欄命名
names(hlt)[1:2] <- c('性別','年齡')
#hlt
#不同性別與年齡的健康功能圖
par(family='Heiti TC Light')
with(hlt, dotchart(功能, group =性別, labels = 年齡, xlab = '功能'))
